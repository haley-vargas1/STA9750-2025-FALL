Acquiring the Data

Firstly, we will need to acquire the data from Netflix’s Tudum Top 10. Specifically, we will be analyzing two separate data files:

  • Global Top 10
  • Country-wide Top 10
Show the code 
if(!dir.exists(file.path("data", "mp01"))){
    dir.create(file.path("data", "mp01"), showWarnings=FALSE, recursive=TRUE)
}

GLOBAL_TOP_10_FILENAME <- file.path("data", "mp01", "global_top10_alltime.csv")

if(!file.exists(GLOBAL_TOP_10_FILENAME)){
    download.file("https://www.netflix.com/tudum/top10/data/all-weeks-global.tsv", 
                  destfile=GLOBAL_TOP_10_FILENAME)
}

COUNTRY_TOP_10_FILENAME <- file.path("data", "mp01", "country_top10_alltime.csv")

if(!file.exists(COUNTRY_TOP_10_FILENAME)){
    download.file("https://www.netflix.com/tudum/top10/data/all-weeks-countries.tsv", 
                  destfile=COUNTRY_TOP_10_FILENAME)
}

Task 2: Data Cleaning

If we notice carefully, we can see that when “season_title” is missing, R is reading these values as strings. To fix this, we will need to use the mutate command.

Show the code 
GLOBAL_TOP_10 <- GLOBAL_TOP_10 |>
  mutate(season_title = if_else(season_title == "N/A", NA_character_, season_title))

Question 8: The movie Red Notice has a runtime of 1 hour and 58 minutes. Approximately how many views did it receive in 2021? Note that Netflix does not provide the weekly_views values that far back in the past, but you can compute it yourself using the total view time and the runtime.

Answer: Red Notice received approximately 201,732,203 views in 2021.

Show the code 
library(lubridate)
library(scales)

# runtime in hours
runtime_hours <- 118 / 60  # 118 minutes = 1.9666667 hours

red_notice_hours_2021 <- GLOBAL_TOP_10 %>%
  filter(show_title == "Red Notice", year(week) == 2021) %>%
  summarize(total_hours = sum(weekly_hours_viewed, na.rm = TRUE)) %>%
  pull(total_hours)

approx_views_2021 <- red_notice_hours_2021 / runtime_hours
print(approx_views_2021)
[1] 201732203

Question 9: How many Films reached Number 1 in the US but did not originally debut there? What is the most recent film to pull this off?

Answer:

Show the code 
us_films <- COUNTRY_TOP_10 %>%
  filter(category == "Films", country_iso2 == "US") %>%
  group_by(show_title) %>%
  mutate(hit_number_one = any(weekly_rank == 1, na.rm = TRUE)) %>%
  ungroup()

# Were in top 10, rose to #1
late_risers_count <- us_films %>%
  filter(hit_number_one) %>%
  group_by(show_title) %>%
  arrange(week) %>%
  summarise(first_rank = first(weekly_rank), .groups = "drop") %>%
  filter(first_rank > 1) %>%
  nrow()

# Most recent film
most_recent_late_riser <- us_films %>%
  filter(hit_number_one) %>%
  group_by(show_title) %>%
  arrange(week) %>%
  mutate(first_week_at_1 = min(week[weekly_rank == 1], na.rm = TRUE)) %>%
  summarise(
    debut_week = first(week),
    debut_rank = first(weekly_rank),
    week_hit_1 = first(first_week_at_1),
    .groups = "drop"
  ) %>%
  filter(debut_rank > 1) %>%
  slice_max(debut_week, n = 1) %>%
  pull(show_title)

# Step 4: Print results
cat("🎬 Number of US films that debuted below #1 but later reached #1:", late_risers_count, "\n")

Question 10: Which TV show/season hit the top 10 in the most countries in its debut week? In how many countries did it chart?

Answer: Emily in Paris: Season 2 hit the top 10 in 94 countries in its debut week.

Show the code 
tv_debut_countries <- COUNTRY_TOP_10 |>
  # 1. Focus only on TV content
  filter(startsWith(category, "TV")) |>
  
  # 2. Find the debut week per show/season
  group_by(show_title, season_title) |>
  mutate(debut_week = min(week, na.rm = TRUE)) |>
  ungroup() |>
  
  # 3. Keep only records from that debut week
  filter(week == debut_week) |>
  
  # 4. Count how many unique countries it appeared in during debut week
  group_by(show_title, season_title) |>
  summarise(num_countries = n_distinct(country_name), .groups = "drop") |>
  
  # 5. Find the top result
  slice_max(order_by = num_countries, n = 1)

print(tv_debut_countries)
# A tibble: 1 × 3

  <chr>          <chr>                            <int>
1 Emily in Paris Emily in Paris: Season 2            94
